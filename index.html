TIME_data <- readxl::read_xlsx("data/Cercospora_TIME_measure.xlsx","TIME")
TIME_desc <- readxl::read_xlsx("data/Cercospora_TIME_description.xlsx","TIME")
TIME_data_disease <-  TIME_data
TIME_data_yield   <-  TIME_desc
  
n_lesion_plot <- TIME_data_disease %>%
  filter(lesion_n > 0) %>%
  group_by(year, trial_ID, plot) %>%
  summarise(
    CPB = mean(CPB, na.rm = TRUE),
    n_lesion = n(),
    .groups = "drop",
    inc  = (n_lesion * 100)/50)

TIME_yield_data <- TIME_data_yield %>% 
  mutate(
     yld = weight_12/(80/43560), 
     kg_ha = yld*1.2085) %>% 
  dplyr::select(year, trial_ID, trial, env, field, plot, blk, trt, 
                  gen,fung, fung_time, total, head, yld, kg_ha, NBLS)

CERC_TIME_data <- right_join(TIME_yield_data, n_lesion_plot,
                  by = c("year","trial_ID","plot")) %>%
                  filter(!field %in% c("WIN")) %>%
                  filter(!fung %in% c("Tilt+A.Top")) %>%
  mutate(
    pressure = ifelse(inc > median(inc), "high", "low")) %>% 
  filter(!is.na(trial))
